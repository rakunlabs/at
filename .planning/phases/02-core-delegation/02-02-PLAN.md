```go
// Fire async delegation in background goroutine.
// Uses context.Background() because the HTTP request context will be cancelled
// after the 202 response is sent, but delegation may take 15+ seconds.
go func() {
    delegCtx := context.Background()
    if err := s.runOrgDelegation(delegCtx, org, record, org.HeadAgentID, 0); err != nil {
        slog.Error("org-delegation: failed",
            "org_id", org.ID,
            "task_id", record.ID,
            "error", err,
        )
        // Update task status to reflect failure.
        if s.taskStore != nil {
            _ = s.taskStore.UpdateTask(delegCtx, record.ID, service.Task{
                Status: service.TaskStatusCancelled,
                Result: fmt.Sprintf("delegation failed: %v", err),
            })
        }
    }
}()
```

Also add `"context"` to the imports if not already present (check — it likely already is used for `r.Context()` calls but `context.Background()` needs the package import).

IMPORTANT: The variable `org` (type `*service.Organization`) and `record` (type `*service.Task`) are already in scope at the point where the placeholder comment lives. `org` is loaded on line 45, `record` on line 120.
